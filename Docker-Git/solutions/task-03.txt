FROM node:24-alpine

WORKDIR /app

RUN npm install

COPY . .

EXPOSE 3000

CMD ["node", "app.js"]


ALL DOCKER COMMANDS TO RUN:
----------------------------

1. docker build -t my-node-app:v1 .
   - Builds Docker image from Dockerfile with tag my-node-app:v1

2. docker images
   - Lists all Docker images to verify build

3. docker run -d -p 3000:3000 --name my-node-container my-node-app:v1
   - Runs container in detached mode (-d)
   - Maps port 3000 from container to host (-p 3000:3000)
   - Names the container my-node-container

4. docker ps
   - Lists all running containers

5. docker inspect my-node-container
   - Shows detailed information about the container

6. curl http://localhost:3000
   - Tests the application (should return "Hello Docker!")

7. docker logs my-node-container
   - Views container logs

8. docker stop my-node-container
   - Stops the running container

9. docker rm my-node-container
   - Removes the stopped container

10. docker rmi my-node-app:v1
    - Removes the Docker image (optional cleanup)

DOCKERFILE EXPLANATION:
-----------------------

FROM node:24-alpine:
  Uses Node.js 24 Alpine Linux base image (lightweight)

WORKDIR /app:
  Sets working directory inside container to /app

COPY package.json ./:
  Copies package.json to container (for dependency installation)

RUN npm install:
  Installs Node.js dependencies inside container

COPY . .:
  Copies all application files to container

EXPOSE 3000:
  Documents that container listens on port 3000

CMD ["node", "app.js"]:
  Defines default command to run when container starts
